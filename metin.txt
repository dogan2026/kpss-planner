<script>
// ============== VERİ BANKASI ==============
const quotes = [
    {q:'Hayatta en hakiki mürşit ilimdir.', p:'Atatürk'},
    {q:'Matematik, aklın müziğidir.', p:'James Joseph Sylvester'},
    {q:'Eğitim, insanın içindeki ateşin yakılmasıdır.', p:'Socrates'},
    {q:'Başarı, hazırlık fırsatla buluştuğunda ortaya çıkar.', p:'Seneca'},
    {q:'Bugünün çalışması, yarının başarısını hazırlar.', p:'Atatürk'}
];

const subjects = [
    {label:'➗ Matematik', topics:['Sayılar','Problemler','Geometri'], color:'#3b82f6'},
    {label:'📚 Türkçe', topics:['Paragraf','Dil Bilgisi','Yazım Kuralları'], color:'#10b981'},
    {label:'🏺 Tarih', topics:['Osmanlı Tarihi','İnkılap Tarihi','Selçuklular'], color:'#f59e0b'},
    {label:'🌍 Coğrafya', topics:['Türkiye Fiziki Coğrafyası','Nüfus ve Yerleşme'], color:'#6366f1'}
];

// ============== DİNAMİK MANTIK ==============
function updateDashboard() {
    const d = new Date();
    const dayOfMonth = d.getDate(); // Ayın günü (1-31)
    
    // 1. Günün Sözünü Seç
    const quote = quotes[dayOfMonth % quotes.length];
    document.getElementById('quote-text').innerText = `"${quote.q}"`;
    document.getElementById('quote-author').innerText = `- ${quote.p}`;

    // 2. Günün Dersini ve Konusunu Seç
    const sub = subjects[dayOfMonth % subjects.length];
    const topic = sub.topics[d.getDay() % sub.topics.length]; // Haftanın gününe göre alt konu
    
    document.getElementById('task-subject').innerText = sub.label;
    document.getElementById('task-detail').innerText = "Çalışılacak: " + topic;
    document.querySelector('#daily-task .task-item').style.borderLeftColor = sub.color;

    // 3. Mini Deneme İçeriği (Her gün farklı soru dağılımı)
    const soruSayisi = (dayOfMonth % 3 + 1) * 10; // 10, 20 veya 30 soru
    document.getElementById('exam-detail').innerText = `${sub.label} branşından ${soruSayisi} soruluk mini deneme çöz!`;
}

// Sayfa yüklendiğinde çalıştır
window.onload = updateDashboard;
</script>
